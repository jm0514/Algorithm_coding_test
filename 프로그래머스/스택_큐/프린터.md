```java
import java.util.LinkedList;
import java.util.Queue;

class Solution {
    public int solution(int[] priorities, int location) {
        int answer = 0;
        Queue<Print> Q = new LinkedList<>();
        
        for(int i = 0; i < priorities.length; i++){
            Q.offer(new Print(priorities[i],i)); //큐에 Print 객체를 생성하고 우선순위와 인덱스를 삽입
        }
        
        while(!Q.isEmpty()){
            Print print = Q.poll();//Print 클래스 참조
            
            boolean maxPriority = false; // 제일 큰 수를 판단 
		    for(Print sub : Q){
			    if(print.priority < sub.priority) {
				    maxPriority = true;
				break;
			    }
		    }   

		if(maxPriority){
			Q.add(print);
        }else{
			answer++;
			if(print.index == location){
				break;
            }
		    }
        }
        
        return answer;
    }
    
    public class Print{ // 우선순위와 인덱스를 따로 클래스로 구현
        private int priority;
        private int index;
        
        public Print(int priority, int index){
            this.priority=priority;
            this.index=index;
        }
    }
}
```
