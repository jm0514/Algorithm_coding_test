# 프로그래머스 실패율
[프로그래머스 실패율](https://school.programmers.co.kr/learn/courses/30/lessons/42889)
```java
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

class Solution {
    public int[] solution(int N, int[] stages) {
        int nPlayers = stages.length;
        int[] nStagePlayers = new int[N + 2];
        for (int s : stages) {
            nStagePlayers[s] += 1;
        }

        int remainingPlayers = nPlayers;
        List<Fail> stages1 = new ArrayList<>();
        for (int id = 1 ; id <= N; id++) {
            double failure = (double) nStagePlayers[id] / remainingPlayers;
            remainingPlayers -= nStagePlayers[id];

            Fail s = new Fail(id, failure);
            stages1.add(s);
        }
        Collections.sort(stages1, Collections.reverseOrder());

        int[] answer = new int[N];
        for (int i = 0; i < N; i++) {
            answer[i] = stages1.get(i).id;
        }
        return answer;
    }

    class Fail implements Comparable<Fail> {
        public int id;
        public double failure;

        public Fail(int id, double failure) {
            this.id = id;
            this.failure = failure;
        }

        @Override
        public int compareTo(Fail o) {
            if (failure < o.failure ) {
                return -1;
            }
            if (failure > o.failure ) {
                return 1;
            }
            return 0;
        }
    }
}
```
