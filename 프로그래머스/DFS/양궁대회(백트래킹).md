# 프로그래머스 양궁대회
[프로그래머스 양궁대회](https://school.programmers.co.kr/learn/courses/30/lessons/92342)
```java
import java.util.*;

class Solution {
    static int[] apeach = new int[11];
    static int[] lion = new int[11];
    static int[] answer = new int[11];
    static int max = 0;
    public int[] solution(int n, int[] info) {
        apeach = info;
        backtracking(n, 0, 0);
        if(Arrays.stream(answer).filter(e -> e == 0).count() == 11) {
        	answer = new int[]{-1};
        }
        return answer;
    }
    
    private static void backtracking(int n, int idx, int cnt) {
        
        if (cnt == n) {
            int LScore = 0;
            int AScore = 0;
            
            for (int i = 0; i <= 10; i++) {
                if (lion[i] == 0 && apeach[i] == 0) continue;
                if (lion[i] > apeach[i]) LScore += 10 - i;
                else AScore += 10 - i;
            }
            
            int nowScore = LScore - AScore;
            
            if (nowScore > 0) {
                if (nowScore > max) {
                    max = nowScore;
                    answer = lion.clone();
                } else if (nowScore == max) {
                    for (int i = 10; i >= 0; i--) {
                        if (lion[i] > answer[i]) {
                            answer = lion.clone();
                        } else if (lion[i] < answer[i]) return;
                    }
                }
            }
            
            return;
        }
        
        for (int i = idx; i <= 10; i++) {
            lion[i]++;
            backtracking(n, idx, cnt + 1);
            lion[i]--;
        }
    }
}
```
