# 프로그래머스 공원 산책
[프로그래머스 공원 산책](https://school.programmers.co.kr/learn/courses/30/lessons/172928)
```java
import java.util.*;

class Solution {
    
    static class Point{
        int x, y;
        public Point(int x, int y){
            this.x = x;
            this.y = y;
        }
    }
    
    static char[][] board;
    static boolean[][] check;
    static int X, Y;
    static int[] dx = {-1, 0, 1, 0}; // 북N 동E 남S 서W
    static int[] dy = {0, 1, 0, -1};
    static int[] answer;
    
    public int[] solution(String[] park, String[] routes) {
        answer = new int[2];
        Y = park.length;
        X = park[0].length();
        board =  new char[Y][X];
        check = new boolean[Y][X];
        
        for(int i = 0; i < Y; i++) {
            for(int j = 0; j < X; j++) {
                board[i][j] = park[i].charAt(j);
            }
        }
        
        for(int i = 0; i < Y; i++){
            for(int j = 0; j < X; i++){
                if(board[i][j] == 'S'){
                    BFS(i, j, routes);
                    System.out.print("i" + i +  "j" + j);
                    break;
                }
            }
        }
        
        return answer;
    }
    
    public static void BFS(int y, int x, String[] routes) {
        Queue<Point> Q = new LinkedList<>();
        check[y][x] = true;
        Q.add(new Point(x, y));
        int index = 0;
        
        while(!Q.isEmpty()){
            Point point = Q.poll();
            int nowX = point.x;
            int nowY = point.y;
            if(index == routes.length){
                answer[0] = nowX;
                answer[1] = nowY;
                break;
            } 
            
            index++;
            int nextX = 0, nextY = 0;
            String[] sp = routes[index].split(" ");
            int move = Integer.parseInt(sp[1]);
            char ch = sp[0].charAt(0);
            
            if(ch == 'E'){ //1
                while(move --> 0){
                    nextX = nowX + dx[1];
                    nextY = nowY + dx[1];
                }
            } else if(ch == 'S'){ //2
                while(move --> 0){
                    nextX = nowX + dx[2];
                    nextY = nowY + dx[2];
                }
            } else if(ch == 'W'){ //3
                while(move --> 0){
                    nextX = nowX + dx[3];
                    nextY = nowY + dx[3];
                }
            } else if(ch == 'N') { // N 0
                while(move --> 0){
                    nextX = nowX + dx[0];
                    nextY = nowY + dx[0];
                }
            }
            
            if(nextY < 0 || nextY >= Y || nextX < 0 || nextX >= X) continue;
            if(!check[nextY][nextX] && board[nextY][nextX] == 'O'){
                Q.add(new Point(nextX, nextY));
                check[nextY][nextX] = true;
                System.out.print("nextX: " + nextX + ", " + "nextY : " + nextY);
            }
            
        }
    }
}
```
