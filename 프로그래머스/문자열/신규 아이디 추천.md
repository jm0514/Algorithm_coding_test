# 프로그래머스 신규 아이디 추천
[프로그래머스 신규 아이디 추천](https://school.programmers.co.kr/learn/courses/30/lessons/72410)
```java
import java.util.*;

class Solution {
    public String solution(String new_id) {
        String answer = "";
        
        String step1 = new_id.toLowerCase();
        
        String step2 = step1.replaceAll("[^0-9a-z-_.]", "");
        String step3 = step2.replaceAll("[.]{2,}", ".");
        String step4 = step3.replaceAll("^[.]{1}", "");
        
        if(step4.equals("")) step4 += "a";
        String step5 = step4;
        if(step5.length() >= 15){
            step5.substring(0,15);
        }
        String step6 = step5.replaceAll("[.]{1}$", "");
        
        String step7 = step6;
        if(step7.length() <= 2){
            for(int i = step7.length(); i < 3; i++){ 
                step7 += step7.substring(step7.length()-1);
            } 
        }
        answer = step7;
        return answer;
    }
}
```
