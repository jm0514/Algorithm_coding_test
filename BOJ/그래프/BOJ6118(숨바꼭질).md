# 백준 6118번 숨바꼭질
[백준 6118번](https://www.acmicpc.net/problem/6118)
```java
import java.util.*;
import java.io.*;

class Main {
    static class Node {
        int now, next;
        Node(int now, int next) {
            this.now = now;
            this.next = next;
        }
    }
    static int N, M, dis, count, end;
    static ArrayList<Integet> list[];
    static boolean check[];
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());
        
        list = new ArrayList[N + 1];
        check = new boolean[N + 1];
        for (int i = 0; i <= N; i++) {
            list[i] = new ArrayList<>();
        }
        for (int i = 0; i < M; i++) {
            st = new StringTokenizer(br.readLine());
            int a = Integer.parseInt(st.nextToken());
            int b = Integer.parseInt(st.nextToken());
            list[a].add(b);
            list[b].add(a);
        }
        
        BFS();
    }
    
    private static void BFS() {
        Queue<Node> Q = new LinkedList<>();
        check[1] = true;
        Q.add(new Node(1, 0));

        while (!Q.isEmpty()) {
            Node n = Q.poll();
            int now = n.now;
            int next = n.next;

            if (next > dis) {
                dis = next;
                end = now;
                count = 1;
            } else if (next == dis) {
                if (dis > now) {
                    end = now;
                }
                count++;
            }

            for (int i = 0; i < list[now].size(); i++) {
                int next_next = list[now].get(i);
                if (!check[next_next]) {
                    check[next_next] = true;
                    Q.offer(new Node(next_next, next + 1));
                }
            }
        }
    } 
}
```
