# 백준 16235번 나무 재테크
[백준 16235번](https://www.acmicpc.net/problem/16235)
```java
import java.util.*;
import java.io.*;

class Main {
    static class Tree implements Comparable<Tree> {
        int x, y, age;
        Tree(int x, int y, int age){
            this.x = x;
            this.y = y;
            this.age = age;
        }

        @Override
        public int compareTo(Tree o) {
            return this.age - o.age;
        }
    }
    static final int START_NUM = 5;
    static final int TREE_INFO_SIZE = 3;
    static int[] dx = {-1, 1, 0, 0, -1, 1, -1, 1};
    static int[] dy = {0, 0, -1, 1, -1, -1, 1, 1};
    static int[][] land, food;
    static int N;
    static Queue<Tree> treeQ = new LinkedList<>();
    static List<Tree> deadTree;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        N = Integer.parseInt(st.nextToken());
        int M = Integer.parseInt(st.nextToken());
        int K = Integer.parseInt(st.nextToken());
        food = new int[N][N];
        land = new int[N][N];

        for (int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < N; j++) {
                food[i][j] = Integer.parseInt(st.nextToken());
                land[i][j] = START_NUM;
            }
        }

        for (int i = 0; i < M; i++) {
            st = new StringTokenizer(br.readLine(), " ");
            int x = Integer.parseInt(st.nextToken());
            int y = Integer.parseInt(st.nextToken());
            int age = Integer.parseInt(st.nextToken());
            treeQ.add(new Tree(x - 1, y - 1, age));
        }

        Collections.sort((List<Tree>)treeQ);

        while(K > 0) {
            spring();
            summer();
            fall();
            winter();
            K--;
        }

        System.out.println(treeQ.size());

    }

    static void spring() {
        int size = treeQ.size();
        deadTree = new ArrayList<>();
        while(size > 0) {
            Tree tree = treeQ.poll();
            if (tree.age <= land[tree.x][tree.y]) {
                land[tree.x][tree.y] -= tree.age;
                treeQ.add(new Tree(tree.x, tree.y, tree.age + 1));
            } else {
                deadTree.add(new Tree(tree.x, tree.y, tree.age));
            }
            size--;
        }
    }

    static void summer() {
        for (Tree tree : deadTree) {
            land[tree.x][tree.y] += tree.age / 2;
        }
    }

    static void fall() {
        // 큐에서 poll로 없어진 나무들의 정보를 저장하는 저장소
        ArrayList<Tree> parent = new ArrayList<>();
        int size = treeQ.size();
        while (size > 0) {
            Tree tree = treeQ.poll();
            parent.add(tree);
            if (tree.age % 5 == 0) {
                for (int i = 0; i < 8; i++) {
                    int nx = dx[i] + tree.x;
                    int ny = dx[i] + tree.y;
                    if (nx < 0 || ny < 0 || nx >= N || ny >= N) continue;
                    treeQ.add(new Tree(nx, ny, 1));
                }
            }
            size--;
        }

        for (Tree tree :parent) {
            treeQ.add(tree);
        }
    }

    static void winter() {
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                land[i][j] += food[i][j];
            }
        }
    }
    
}
```
