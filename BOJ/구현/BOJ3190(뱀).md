# 백준 3190번 뱀
[백준 3190번](https://www.acmicpc.net/problem/3190)
```java
import java.util.*;
import java.io.*;

class Main {

    static class Point {
        int x, y;
        private Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
    // 오른쪽, 아래, 왼쪽, 위쪽
    static int[] dx = {1, 0, -1 , 0};
    static int[] dy = {0, 1, 0 , -1};
    static int N, dir = 0, SX = 0, SY = 0, count = 0;
    static int tailX = 0, tailY = 0;
    static int[][] board, snake;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        board = new int[N][N];
        snake = new int[N][N];
        snake[0][0] = 1;
        int K = Integer.parseInt(br.readLine());

        StringTokenizer st;
        for (int i = 0; i < K; i++) {
            st = new StringTokenizer(br.readLine(), " ");
            int Y = Integer.parseInt(st.nextToken());
            int X = Integer.parseInt(st.nextToken());
            board[Y - 1][X - 1] = 2; // 2는 사과
        }

        int L = Integer.parseInt(br.readLine());
        for (int i = 0; i < L; i++) {
            st = new StringTokenizer(br.readLine(), " ");
            int x = Integer.parseInt(st.nextToken());
            char c = st.nextToken().charAt(0);
            if(!BFS(x, c)) break;
        }

        System.out.println(count);
    }

    private static boolean BFS(int x, int c) {
        Queue<Point> Q = new LinkedList<>();
        Q.offer(new Point(SX, SY));
        Point p = Q.poll();
        int nowX = p.x;
        int nowY = p.y;
        for (int i = 0; i < x; i++) {
            int nextX = nowX + dx[dir];
            int nextY = nowY + dy[dir];
            count++;
            if (nextY < 0 || nextX < 0 || nextY >= N || nextX >= N
                    || snake[nextY][nextX] == 1) {
                return false;
            }

            snake[nextY][nextX] = 1;
            if (board[nextY][nextX] != 2) {
                snake[tailY][tailX] = 0;
                tailX = nextX;
                tailY = nextY;
            }

            nowX = nextX;
            nowY = nextY;
            SX = nextX;
            SY = nextY;
        }

        if(dir == 0) {
            if(c == 'L') dir = 3;
            else if(c == 'D') dir = 1;
        } else if(dir == 1 || dir == 3) {
            if(c == 'L') dir = 2;
            else if(c == 'D') dir = 0;
        }  else if(dir == 2) {
            if(c == 'L') dir = 1;
            else if(c == 'D') dir = 3;
        }
        return true;
    }
}
```
