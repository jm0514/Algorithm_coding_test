# 백준 2531번 회전 초밥
[백준 2531번](https://www.acmicpc.net/problem/2531)
```java
import java.io.*;
import java.util.*;

class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        int N = Integer.parseInt(st.nextToken());
        int d = Integer.parseInt(st.nextToken());
        int k = Integer.parseInt(st.nextToken());
        int c = Integer.parseInt(st.nextToken());
        // 스시 종류를 담은 배열
        int[] sushi = new int[N];
        // 먹은 스시를 카운트하는 배열(카운트의 총합은 k이다.)
        int[] eat = new int[d + 1];

        for (int i = 0; i < N; i++) {
            sushi[i] = Integer.parseInt(br.readLine());
        }

        int count = 0;
        // 처음 k개를 연속으로 먹는 경우
        for (int i = 0; i < k; i++) {
            // 초밥의 가짓수를 구하는 것이기 떄문에 0일 경우만 count한다.
            if (eat[sushi[i]] == 0) {
                count++;
            }
            // 먹은 스시를 체크한다.
            eat[sushi[i]]++;
        }

        // 처음 연속으로 k개를 연속으로 먹은 경우의 최대 값
        int maxLen = count;
        // 투 포인터를 이용하기 때문에 시작점을 N까지 순회한다.
        for (int start = 1; start < N; start++) {
            if (maxLen <= count) {

                if (eat[c] == 0) {
                    maxLen = count + 1;
                } else {
                    maxLen = count;
                }
            }

            int end = (start + k - 1) % N;
            if (eat[sushi[end]] == 0) {
                // 가짓수 세기
                count++;
            }
            // 먹은 스시 체크
            eat[sushi[end]]++;
            // 배열을 이동해서 개수를 빼준다.
            eat[sushi[start - 1]]--;
            if (eat[sushi[start - 1]] == 0) {
                // 가짓수가 0이 된 경우 카운트에서 뺀다.
                count--;
            }
        }

        System.out.println(maxLen);
    }
}
```
