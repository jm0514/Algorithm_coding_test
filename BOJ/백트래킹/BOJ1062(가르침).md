# 백준 1062번 가르침
[백준 1062번](https://www.acmicpc.net/problem/1062)
```java
import java.io.*;
import java.util.*;

class Main {
    static String[] arr;
    static boolean[] check;
    static int answer = 0;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        int N = Integer.parseInt(st.nextToken());
        int K = Integer.parseInt(st.nextToken());
        arr = new String[N];

        check = new boolean[26];
        check[0]=true; // a
        check[2]=true; // c
        check[8]=true; // i
        check[13]=true; // n
        check[19]=true; // t

        for (int i = 0; i < N; i++) {
            String input = br.readLine();
            arr[i] = input.substring(4, input.length() - 4);
        }

        dfs(0, 0, K - 5);
        System.out.println(answer);
    }

    private static void dfs(int start, int depth, int max) {
        if (depth == max) {
            int count = 0;
            for (int i = 0; i < arr.length; i++) {
                boolean result = true;
                for (int j = 0; j < arr[i].length(); i++) {
                    int temp = arr[i].charAt(j) - '0';
                    if (!check[temp]) {
                        result = false;
                        break;
                    }
                }
                if (result) count++;
            }

            Math.max(answer, count);
            return;
        }

        for (int i = start; i < check.length; i++) {
            if (i == 0 || i == 2 || i == 8 || i == 13 || i == 19) continue;
            if (!check[i]) {
                check[i] = true;
                dfs(start + 1, depth + 1, max);
                check[i] = false;
            }
        }
    }
}
```
