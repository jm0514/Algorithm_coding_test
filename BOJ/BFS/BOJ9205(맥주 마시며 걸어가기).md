# 백준 9205번 맥주 마시면서 걸어가기
[백준 9205](https://www.acmicpc.net/problem/9205)
```java
import java.io.*;
import java.util.*;

public class Main6 {

    static public class Point {
        int x;
        int y;

        public Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

    static int t;
    static int n;
    static Point[] points;
    static boolean[] check;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringBuilder sb = new StringBuilder();
        StringTokenizer st;

        t = Integer.parseInt(br.readLine());
        while (t -- > 0) {
            n = Integer.parseInt(br.readLine());
            points = new Point[n + 2];
            for (int i = 0; i < n + 2; i++) {
                st = new StringTokenizer(br.readLine());
                points[i] = new Point(Integer.parseInt(st.nextToken()), Integer.parseInt(st.nextToken()));
            }

            if (!bfs()) {
                sb.append("sad");
            } else {
                sb.append("happy");
            }
            sb.append('\n');
        }

        System.out.print(sb);
    }

    public static boolean bfs() {
        check = new boolean[n + 2];
        Queue<Point> que = new LinkedList<>();
        que.add(new Point(points[0].x, points[0].y));
        check[0] = true;

        while (!que.isEmpty()) {
            Point cur = que.poll();

            for (int i = 1; i < n + 2; i++) {
                if (check[i]) continue;
                if (!passable(cur, points[i])) continue;
                if (i == n + 1) return true;

                que.add(new Point(points[i].x, points[i].y));
                check[i] = true;
            }
        }
        return false;
    }

    static public boolean passable(Point a, Point b) {
        int distance = Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
        if (distance <= 1000) {
            return true;
        } else {
            return false;
        }
    }

}
```
