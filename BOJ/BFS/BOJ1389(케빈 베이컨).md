# 백준 1389번 케빈 베이컨의 6단계 법칙
[백준 1389번](https://www.acmicpc.net/problem/1389)
```java
import java.util.*;
import java.io.*;

class Main{

    static class Info{
        int idx, count;
        public Info(int idx, int count){
            this.idx = idx;
            this.count = count;
        }
    }
    static int minCount = Integer.MAX_VALUE;
    static int minIdx = 0;
    static int N, M;
    static int[][] arr;
    static boolean[] check;

    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        st = new StringTokenizer(br.readLine(), " ");
        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());
        arr = new int[N + 1][N + 1];

        for(int i = 0; i < M; i++){
            st = new StringTokenizer(br.readLine());
            int A = Integer.parseInt(st.nextToken());
            int B = Integer.parseInt(st.nextToken());

            arr[A][B] = arr[B][A] = 1;
        }

        for(int i = 1; i <= N; i++){
            check = new boolean[N + 1];
            BFS(i);
        }
        System.out.println(minIdx);
    }

    public static void BFS(int first){
        Queue<Info> Q = new LinkedList<>();
        Q.offer(new Info(first, 0));
        check[first] = true;
        int sum = 0;

        while(!Q.isEmpty()){
            Info info = Q.poll();
            sum += info.count;
            for(int j = 1; j <= N; j++){
                if(!check[j] && arr[info.idx][j] == 1){
                    check[j] = true;
                    Q.offer(new Info(j, info.count + 1));
                }
            }
        }

        if(minCount > sum) { // 제일 작은 합
            minCount = sum;
            minIdx = first;
        }else if(minCount == sum){ // 비교 대상이 같은 경우
            minIdx = Math.min(first, minIdx); //작은 인덱스 반환
        }

    }
}
```
