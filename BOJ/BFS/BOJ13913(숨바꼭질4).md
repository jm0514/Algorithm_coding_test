# 백준 13913번 숨바꼭질 4
[백준 13913번](https://www.acmicpc.net/problem/13913)
```java
import java.util.*;
import java.io.*;

class Main {
    static class Point{
        int x, count;
        public Point(int x, int count) {
            this.x = x;
            this.count = count;
        }
    }
    static int N, dong, min;
    static int[] dx = {1, -1, 2};
    static int[] arr, trace;
    static boolean[] check;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        int su = Integer.parseInt(st.nextToken());
        dong = Integer.parseInt(st.nextToken());
        int max = Math.max(su, dong);
        N = max;
        min = Integer.MAX_VALUE;
        arr = new int[max + 2];
        trace = new int[max + 2];
        check = new boolean[max + 2];

        BFS(su);

        System.out.println(min);

        for (int i = 0; i <= N; i++) {
            trace[i] = -1;
        }

        int idx = dong;
        Stack<Integer> stack = new Stack<>();
        stack.add(dong);
        while (trace[idx] != -1) {
            stack.add(trace[idx]);
            idx = trace[idx];
        }
        
        StringBuilder sb = new StringBuilder();
        while(!stack.isEmpty()) {
            sb.append(stack.pop() + " ");
        }

        System.out.print(sb.toString());
    }

    private static void BFS(int su) {
        Queue<Point> Q = new LinkedList<>();
        check[su] = true;
        Q.add(new Point(su, 0));

        while(!Q.isEmpty()) {
            Point p = Q.poll();
            if (p.x == dong) {
                min = p.count;
                return;
            }
            for (int i = 0; i < 3; i++) {
                int np;
                if (i == 2) {
                    np = p.x * dx[2];
                } else {
                    np = p.x + dx[i];
                }
                if (np <= 0 || np > N) continue;
                if (!check[np]) {
                    check[np] = true;
                    Q.add(new Point(np, p.count + 1));
                    trace[np] = p.x;
                }
            }
        }
    }
}
```
