# 백준 14503번 로봇 청소기
[백준 14503번](https://www.acmicpc.net/problem/14503)
```java
import java.util.*;
import java.io.*;

class Main {

    static int[] dx = { -1, 0, 1, 0}; //서, 북, 동, 남(시계 방향)
    static int[] dy = { 0, 1, 0, -1};
    static int[][] arr;
    static int N, M, count;
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        st = new StringTokenizer(br.readLine(), " ");
        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());
        arr = new int[N][M];

        st = new StringTokenizer(br.readLine(), " ");
        int r = Integer.parseInt(st.nextToken());
        int c = Integer.parseInt(st.nextToken());
        int d = Integer.parseInt(st.nextToken()); // 북 0 , 동 1, 남 2, 서 3

        for(int i = 0; i < N; i++){
            st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < M; j++) {
                arr[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        count = 1;
        DFS(r, c, d);
        System.out.print(count);
    }

    public static void DFS(int x, int y, int dir){
        arr[x][y] = -1;
        // 현재 칸의 주변 4칸 중 청소되지 않은 빈 칸이 있는 경우
        for(int i = 0; i < 4; i++){ // 4번 반복(반시계 방향)
            dir = (dir + 3) % 4; // ex) dir 0일 경우 : 3->2->1->0
            int nextX = dx[dir] + x;
            int nextY = dy[dir] + y;
            if(nextX >= 0 && nextX < N && nextY >= 0 && nextY < M && arr[nextX][nextY] == 0){
                count++;
                DFS(nextX, nextY, dir); // 한 칸 전진
                return;// 1번으로 돌아간다
            }
        }
        // 현재 칸이 아직 청소되지 않는 빈칸이 없는 경우(청소돼 있거나(-1), 이동이 가능해야 하기 때문에 벽이 아님(!=1), 앞에서 0은 전부 -1로 처리됨)
        // 후진(180도 방향 전환 후 이동)
        int back = (dir + 2) % 4;
        int backX = x + dx[back];
        int backY = y + dy[back];
        if (backX >= 0 && backX < N && backY >= 0 && backY < M ) {
            if (arr[backX][backY] != 1 && arr[backX][backY] == -1) {
                DFS(backX, backY, dir);
            }
        }

    }


}
```
