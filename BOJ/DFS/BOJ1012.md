# 백준 1012번
[백준 1012번](https://www.acmicpc.net/problem/1012)
```java
import java.io.*;
import java.util.*;

public class Main{
    static StringBuilder sb;
    static int arr[][];
    static int M,N;
    static int dx = {-1, 0, 1, 0};
    static int dy = {0, 1, 0, -1};
    public static void DFS(int x, int y, int arr[][]){
        for(int i = 0; i < 4; i++){
            int nextX = x + dx[i]; //저장된 dx만큼 x축 이동
            int nextY = y + dy[i]; //저장된 dy만큼 y축 이동
            if(nextX >= 0 && nextX < M && nextY >= 0 && nextY < N){
                arr[nextX][nextY] = 0;
                DFS(nextX, nextY, arr);
            }
        }
    }
    
    public static void Count(int arr[][]){
        for(int i = 0; i < M; i++){
            for(int j = 0; j < N; j++){
                if(arr[i][j] == 1){
                    answer++;
                    arr[i][j] = 0; //DFS는 nextX, nextY부터 시작하기 때문에 시작점은 따로 설정
                    DFS(i, j, arr);
                }
            }
        }
        sb.append(answer).append('\n');
    }
    
    public static void main(String[] args)throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;
        
        while(T-->0) {
            int T = Integer.parseInt(br.readLine());
            st = new StringTokenizer(br.readLine(), " ");
            M = Integer.parseInt(st.nextToken()); //가로
            N = Integer.parseInt(st.nextToken()); //세로
            int K = Integer.parseInt(st.nextToken()); //배추 위치 개수
            
            arr = new int[M+1][N+1];
            
            while (K-->0){
                st = new StringTokenizer(br.readLine(), " ");
                int x = Integer.parseInt(st.nextToken());
                int y = Integer.parseInt(st.nextToken());
                arr[x][y] = 1;
            }
        }
        Count(x, y, arr);
        System.out.print(sb);
    }
    
}
```
