# 백준 1620번
[백준 1620번](https://www.acmicpc.net/problem/1620)

```java
import java.util.*;

public class Main{
	
	public static void main(String args[]) {
		Scanner kb = new Scanner(System.in);
		Map<String, String> map = new HashMap<>();
		StringBuffer sb = new StringBuffer();
		int n = kb.nextInt();
		int m = kb.nextInt();
		
		String arr[] = new String[n+1];
		
		for(int i = 0; i<n; i++) {
			String monster = kb.nextLine();
			String num = kb.nextLine();
			
			 map.put(monster, num);
			 map.put(num, monster);
		}
		
		for(int j = 0; j<m; j++) {
			sb.append(map.get(kb.nextLine())).append("\n");
		}

		System.out.println(sb);
	}
}
```
